# Runtime dependencies for Pokémon Collection Manager.
# Pinned to compatible minimum versions; upgrade deliberately.

# GUI framework (Qt for Python) — used from Step 2 onwards.
PySide6>=6.7,<7

# HTTP client for the pokemontcg.io price/catalog provider — used from Step 4.
requests>=2.31,<3

# Windows UI Automation (reads an already-open, normally-launched browser
# window's on-screen text) for the Cardmarket price lookup — used from
# Step 7. Not a browser-automation/scraping library: it never drives or
# navigates the browser, only reads what's already rendered on screen,
# the same mechanism screen readers use. Pulls in pywin32 (win32gui), used
# directly to identify the actual foreground window rather than scanning
# every open window by title. Windows-only: app.pricing.browser dispatches
# to a different backend per sys.platform (see app/pricing/browser/_macos.py
# for the equivalent on macOS, using the Accessibility API instead).
pywinauto>=0.6.9,<0.7 ; sys_platform == "win32"

# macOS equivalent of the above: the Accessibility API (AXUIElement, via
# ApplicationServices) reads Chrome's already-rendered window content the
# same way VoiceOver does, Cocoa/AppKit (NSWorkspace/NSRunningApplication)
# launches Chrome and re-activates this app afterwards, and Quartz provides
# the window-content screenshot (sealed-product/manual-entry photo capture)
# and the Cmd+W keystroke to close the opened tab. Requires the user to
# grant this app Accessibility permission (System Settings > Privacy &
# Security > Accessibility) the first time -- there is no way around that
# prompt, it's how macOS gates this API for every app that uses it.
pyobjc-framework-Cocoa>=10.0 ; sys_platform == "darwin"
pyobjc-framework-ApplicationServices>=10.0 ; sys_platform == "darwin"
pyobjc-framework-Quartz>=10.0 ; sys_platform == "darwin"

# Excel (.xlsx) export — used from Step 11.
openpyxl>=3.1,<4

# PDF export — used from Step 11.
reportlab>=4.0,<5
